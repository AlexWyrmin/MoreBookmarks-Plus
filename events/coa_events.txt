##NEW##

coa.22 = {
	type = character_event
	hidden = yes
	trigger = {
		any_held_title = { this = title:k_anatolia }
	}
	immediate = {
			if = {
				limit = { 		
					dynasty = dynasty:613 
				}
				title:k_anatolia = { set_coa = k_rum } # 
			}
		
		else = {
			title:k_anatolia = { set_coa = k_anatolia }
		}
	}
}
