	}

			limit = {

				name = unavailable_unique_decisions
				target = flag:flag_unified_burgundy_kingdom

			}
		}
	}

		always = yes
	}

		base = 5
		ai_value_modifier = {
			ai_boldness = 1.0
		}
	}
