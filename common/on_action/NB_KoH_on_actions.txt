##For setting the Kingdom Holy Orders

KOH_HOLY_ORDER_GOVERNMENT = {
	effect = {
        if = {
            limit = {
                scope:title = {  
                    NOT = { has_trait = order_member }
                    OR = {
						has_government = holy_order_county_government
         	       }
				}
			}
            add_trait = order_member
        }
	}
}


on_title_gain = {
	effect = {
		if = {
			limit = {
				NOT = { has_trait = order_member }
				has_government = holy_order_county_government
				}
			add_trait = order_member
		}
	}
}
