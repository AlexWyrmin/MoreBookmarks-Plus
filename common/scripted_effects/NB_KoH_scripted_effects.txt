give_assassins_castle = {
	title:b_alamut.province = { 
		if = {
			limit = {
				NOT = { has_building_or_higher = building_temple_03 }
			}
			# If it doesn't have a free building slot, give it one.
			if = {
				limit = { free_building_slots = 0 }
				add_building_slot = 1
			}
			# Add the building.
			add_building = building_temple_03
		}
		# Barracks.
		if = {
			limit = {
				NOT = { has_building_or_higher = building_pastures_01 }
			}
			# If it doesn't have a free building slot, give it one.
			if = {
				limit = { free_building_slots = 0 }
				add_building_slot = 1
			}
			# Add the building.
			add_building = building_pastures_01
		}
		# Military Camps.
		if = {
			limit = {
				NOT = { has_building_or_higher = building_curtain_walls_08 }
			}
			# If it doesn't have a free building slot, give it one.
			if = {
				limit = { free_building_slots = 0 }
				add_building_slot = 1
			}
			# Add the building.
			add_building = building_curtain_walls_08
		}

		if = {
			limit = {
				NOT = { has_building_or_higher = building_military_camps_07 }
			}
			# If it doesn't have a free building slot, give it one.

				limit = { free_building_slots = 0 }
				add_building_slot = 1
			}
			# Add the building.
			add_building = building_military_camps_07
