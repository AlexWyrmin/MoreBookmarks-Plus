# The special variants keyword declares which asset variations you would generally like to use.
# The available requirement types are culture, region, and dlc.
# Each requirement appends to the expected mesh file name.
# The culture suffix "_building_gfx" is entirely optional - mesh files will remove it.
# Including the symbol * in a key indicates that variants should not be generated for that asset.
# If multiple variants are valid for a given province, the later ones in the list are prioritized.
variants: culture=iberian&region=graphical_mediterranean&dlc=the_fate_of_iberia culture=western
wall: wall_1 wall_2
barracks: barracks_1 barracks_2
farms: farm_1* farm_2
